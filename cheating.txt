void nqueens(int n)
{
	int x[max];
	int k;
	x[0]=-1;
	k=0;
	while(k>=0)
	{
		x[k]=x[k]+1;
		while(x[k]<n && !place(x,k))
			x[k]=x[k]+1;
		if(x[k]<n)
			if (k==n-1)
				display(x,n);
			else
			{
				k++; x[k]=-1;
			}
		else
		k--;
	}
}

void insertionSort(int arr[], int n)
{
    int i, key, j;
    for (i = 1; i < n; i++) {
        key = arr[i];
        j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j = j - 1;
        }
        arr[j + 1] = key;
    }
}
